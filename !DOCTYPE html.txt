          const [matchPercentage, setMatchPercentage] = useState(0);

          const videoRef = useRef(null);
          const streamRef = useRef(null);
          const requestRef = useRef(null);

          // Handle input formatting for Aadhar
          const handleAadharChange = (e) => {
            let val = e.target.value.replace(/[^0-9]/g, '');
            if (val.length > 12) val = val.slice(0, 12);
            val = val.replace(/(\d{4})(?=\d)/g, '$1 ');
            setLoginData(prev => ({ ...prev, aadhar: val }));
          };

          // Login Logic
          const handleLogin = (e) => {
            e.preventDefault();
            // Strict Credential Check
            if (loginData.aadhar === "1111 1111 1111" && loginData.pan) {
              setIsAuthenticated(true);
              setCurrentUser(MOCK_USER);
              setPage('home');
            } else { 
              alert("Access Denied. User not found. Please use Aadhar: 1111 1111 1111"); 
            }
          };

          const handleLogout = () => { setIsAuthenticated(false); setCurrentUser(null); setPage('login'); setVoteStep('list'); setLoginData({ aadhar: '', pan: '' }); };
          const startVoting = (candidate) => { if (currentUser.voted) return; setSelectedCandidate(candidate); setVoteStep('face'); setCameraActive(true); };
          const resetVote = () => { setCurrentUser(prev => ({...prev, voted: false})); setSelectedCandidate(null); setVoteStep('list'); };

          // --- FACE DETECTION LOGIC (TensorFlow.js) ---
          useEffect(() => {
            let model = null;
            
            const startCameraAndDetect = async () => {
              try {
                setFaceScanStatus('loading'); // Loading Camera & Model

                // 1. Get Webcam Access
                if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) throw new Error("No Camera API");
                const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" } });
                streamRef.current = stream;
                
                if (videoRef.current) {
                    videoRef.current.srcObject = stream;
                    await new Promise((resolve) => {
                        videoRef.current.onloadedmetadata = () => {
                            videoRef.current.play();
                            resolve();
                        };
                    });
                }

                // 2. Load BlazeFace Model
                if (!window.blazeface) throw new Error("BlazeFace library not loaded");
                model = await window.blazeface.load();
                setFaceScanStatus('scanning');

                // 3. Detection Loop
                const detect = async () => {
                    if (!videoRef.current || !streamRef.current) return;
                    
                    // Estimate faces
                    const returnTensors = false;
                    const predictions = await model.estimateFaces(videoRef.current, returnTensors);

                    if (predictions.length > 0) {
                        // Face found!
                        setFaceScanStatus('detected');
                        
                        // Simulate "Biometric Matching" Phase
                        setTimeout(() => {
                             setFaceScanStatus('matching');
                             
                             // Animate matching percentage
                             let p = 0;
                             const interval = setInterval(() => {
                                 p += Math.floor(Math.random() * 5) + 2;
                                 if (p >= 98) {
                                     p = 98;
                                     clearInterval(interval);
                                     setFaceScanStatus('success');
                                     setTimeout(() => {
                                         setCameraActive(false);
                                         setVoteStep('fingerprint');
                                     }, 1500);
                                 }
                                 setMatchPercentage(p);
                             }, 100);
                        }, 1000);

                    } else {
                        // Keep scanning if no face found yet or not finished
                        if(faceScanStatus !== 'success' && faceScanStatus !== 'matching' && faceScanStatus !== 'detected') {
                           requestRef.current = requestAnimationFrame(detect);
                        }
                    }
                };
                
                detect();

              } catch (err) {
                console.error("Face detection error:", err);
                setFaceScanStatus('error');
                // Fallback simulation if camera fails (e.g. permission denied)
                setTimeout(() => {
                    setCameraActive(false);
                    setVoteStep('fingerprint');
                }, 3000);
              }
            };

            if (voteStep === 'face' && cameraActive) {
                startCameraAndDetect();
            }

            // Cleanup function
            return () => {
                if (streamRef.current) {
                    streamRef.current.getTracks().forEach(track => track.stop());
                    streamRef.current = null;
                }
                if (requestRef.current) {
                    cancelAnimationFrame(requestRef.current);
                }
            };
          }, [voteStep, cameraActive]);

Vote responsibly.</p></div>
                <div onClick={() => setPage('profile')} className="p-1 transition-all duration-300 border-2 rounded-full cursor-pointer border-amber-500/50 hover:border-amber-400 hover:shadow-[0_0_15px_rgba(245,158,11,0.4)] bg-neutral-800 flex items-center justify-center w-14 h-14 hover:scale-110 active:scale-95 group"><User className="w-7 h-7 text-amber-400 group-hover:text-white transition-colors" /></div>

"View Status" : "Proceed to Polls"} <ChevronLeft className="w-4 h-4 ml-2 rotate-180 group-hover:translate-x-1 transition-transform" /></div>

"Select Candidate" : voteStep === 'face' ? "Security Check" : "Final Verification"}</h2></div>

                                        <button onClick={() => startVoting(c)} className="w-full py-4 bg-stone-800 hover:bg-gradient-to-r hover:from-amber-600 hover:to-yellow-600 text-white font-bold rounded-xl transition-all duration-300 flex items-center justify-center gap-3 group/btn shadow-lg">VOTE FOR {c.party.toUpperCase()}<ChevronLeft className="w-4 h-4 rotate-180 opacity-0 -ml-4 group-hover/btn:opacity-100 group-hover/btn:ml-0 transition-all" /></button>
                                    </div>
                                </div>
                            ))}

                    {/* Face Detection Step */}
                    {voteStep === 'face' && (
                        <div className="flex flex-col items-center justify-center py-8 animate-slide-up">

                                {/* Live Camera Feed */}

                                    <div className="absolute top-4 left-4 bg-red-600 text-white text-[10px] font-bold px-2 py-0.5 rounded animate-pulse z-20">LIVE</div>
                                    {cameraActive ? (
                                        <>
                                            <video ref={videoRef} autoPlay playsInline muted className="w-full h-full object-cover transform scale-x-[-1]" />
                                            <div className="absolute inset-0 z-20 pointer-events-none">
                                                {faceScanStatus === 'scanning' && <div className="absolute left-0 w-full h-1 bg-amber-500/60 shadow-[0_0_15px_rgba(245,158,11,0.8)] animate-scan"></div>}
